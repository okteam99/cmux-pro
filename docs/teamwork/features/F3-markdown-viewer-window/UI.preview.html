Swift example

Inline WKWebView is the rendering host. Below is a swift block with highlight.js tokens.

import AppKit

// AC-24 warm open target: <= 200ms
final class MarkdownViewerWindowManager {
    static let shared = MarkdownViewerWindowManager()
    let processPool = WKProcessPool()
    private var controllers: [String: MarkdownViewerWindowController] = [:]

    func showWindow(for filePath: String) {
        if let existing = controllers[filePath] {
            existing.showWindow(nil)
            return
        }
        // ...
    }
}
